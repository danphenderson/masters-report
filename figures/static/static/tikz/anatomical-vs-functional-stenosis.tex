% Anatomy-function discordance schematic.
% Two panels with identical lumen geometry but distinct pressure-loss
% signatures. Panels placed by hand (no \foreach) so the two regimes
% can carry slightly different annotations, and the bottom prose was
% folded into the caption.
\begin{tikzpicture}[fig, scale=1.12, transform shape,
    panel/.style={draw=wallgray!65, rounded corners=3pt, fill=white,
        line width=0.8pt, minimum width=55mm, minimum height=42mm},
    ptitle/.style={font=\footnotesize\bfseries, align=center},
    trace/.style={line width=0.95pt, smooth},
    plabel/.style={font=\footnotesize, align=center}]

  % ----- left panel: anatomically narrow, mild pressure loss -----
  \node[panel] at (-3.2,0) {};
  \node[ptitle, mathblue!75!black] at (-3.2,1.75)
    {anatomically narrow\\functionally mild};
  \begin{scope}[shift={(-4.9,0.75)}, x=1cm, y=1cm]
    \fill[wallgraylight] (0,0.45) -- (1.15,0.45)
      .. controls (1.55,0.22) and (1.85,0.22) .. (2.25,0.45)
      -- (3.4,0.45) -- (3.4,0.66) -- (0,0.66) -- cycle;
    \fill[wallgraylight] (0,-0.45) -- (1.15,-0.45)
      .. controls (1.55,-0.22) and (1.85,-0.22) .. (2.25,-0.45)
      -- (3.4,-0.45) -- (3.4,-0.66) -- (0,-0.66) -- cycle;
    \fill[bloodredlight!75] (0,0.39) -- (1.15,0.39)
      .. controls (1.55,0.16) and (1.85,0.16) .. (2.25,0.39)
      -- (3.4,0.39) -- (3.4,-0.39) -- (2.25,-0.39)
      .. controls (1.85,-0.16) and (1.55,-0.16) .. (1.15,-0.39)
      -- (0,-0.39) -- cycle;
    \draw[centerline] (-0.05,0) -- (3.45,0);
    \draw[flowArrow] (0.25,0) -- (0.95,0);
    \draw[flowArrow] (2.45,0) -- (3.15,0);
  \end{scope}
  \begin{scope}[shift={(-4.95,-1.0)}, x=1cm, y=1cm]
    \draw[axisArrow] (0,0) -- (3.55,0) node[right, plabel] {$t$};
    \draw[axisArrow] (0,0) -- (0,1.15) node[above, plabel] {$p$};
    \draw[trace, bloodred!75!black]
      plot[domain=0:3.25, samples=80] (\x,{0.74+0.22*sin(110*\x)});
    \draw[trace, mathblue!75!black, dashed]
      plot[domain=0:3.25, samples=80] (\x,{0.56+0.20*sin(110*\x-8)});
    \draw[<->, line width=0.65pt, mathblue!85!black]
      (2.9,{0.56+0.20*sin(110*2.9-8)}) -- (2.9,{0.74+0.22*sin(110*2.9)});
    \node[plabel, mathblue!85!black, anchor=west] at (3.02,0.65)
      {small $\Delta p$};
  \end{scope}

  % ----- right panel: matched narrowing, functionally severe -----
  \node[panel] at (3.2,0) {};
  \node[ptitle, bloodred!75!black] at (3.2,1.75)
    {matched narrowing\\functionally severe};
  \begin{scope}[shift={(1.5,0.75)}, x=1cm, y=1cm]
    \fill[wallgraylight] (0,0.45) -- (1.15,0.45)
      .. controls (1.55,0.22) and (1.85,0.22) .. (2.25,0.45)
      -- (3.4,0.45) -- (3.4,0.66) -- (0,0.66) -- cycle;
    \fill[wallgraylight] (0,-0.45) -- (1.15,-0.45)
      .. controls (1.55,-0.22) and (1.85,-0.22) .. (2.25,-0.45)
      -- (3.4,-0.45) -- (3.4,-0.66) -- (0,-0.66) -- cycle;
    \fill[bloodredlight!75] (0,0.39) -- (1.15,0.39)
      .. controls (1.55,0.16) and (1.85,0.16) .. (2.25,0.39)
      -- (3.4,0.39) -- (3.4,-0.39) -- (2.25,-0.39)
      .. controls (1.85,-0.16) and (1.55,-0.16) .. (1.15,-0.39)
      -- (0,-0.39) -- cycle;
    \draw[centerline] (-0.05,0) -- (3.45,0);
    \draw[flowArrow] (0.25,0) -- (0.95,0);
    \draw[flowArrow] (2.45,0) -- (3.15,0);
  \end{scope}
  \begin{scope}[shift={(1.45,-1.0)}, x=1cm, y=1cm]
    \draw[axisArrow] (0,0) -- (3.55,0) node[right, plabel] {$t$};
    \draw[axisArrow] (0,0) -- (0,1.15) node[above, plabel] {$p$};
    \draw[trace, bloodred!75!black]
      plot[domain=0:3.25, samples=80] (\x,{0.78+0.22*sin(110*\x)});
    \draw[trace, mathblue!75!black, dashed]
      plot[domain=0:3.25, samples=80] (\x,{0.22+0.20*sin(110*\x-8)});
    \draw[<->, line width=0.65pt, bloodred!85!black]
      (2.9,{0.22+0.20*sin(110*2.9-8)}) -- (2.9,{0.78+0.22*sin(110*2.9)});
    \node[plabel, bloodred!85!black, anchor=west] at (3.02,0.55)
      {large $\Delta p$};
  \end{scope}
\end{tikzpicture}
