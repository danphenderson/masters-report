% ---------------------------------------------------------------
% figures/static/static/tikz/stenosis-schematic.tex
% Idealized stenosed compliant blood vessel.
% Notation: R_base (healthy baseline radius), R(z)=R_base(1-s(z)) (stenosed),
% R_min, z_star (minimum-radius location), Q(t),
% p_{\mathrm{g},prox}(t), p_{\mathrm{g},dist}(t), Delta p(t).
% ---------------------------------------------------------------
\begin{tikzpicture}[fig, scale=1.0,
    geomLabel/.style={font=\footnotesize, fill=white, fill opacity=0.88,
                      text opacity=1, inner sep=1.2pt},
    tapLabel/.style={font=\footnotesize, text=mathblue, fill=white,
                     fill opacity=0.92, text opacity=1, inner sep=1.2pt},
    shearLabel/.style={font=\footnotesize, text=stenosisOrange, fill=white,
                       fill opacity=0.9, text opacity=1, inner sep=1.4pt,
                       align=left}]
  \def\L{6}                     % vessel length
  \def\R{1.0}                   % reference radius
  \def\za{2.5}                  % asymmetry-center parameter
  \def\zs{3.4}                  % axial-shift parameter
  \def\sigmaA{1.0}              % asymmetry width
  \def\kapA{0.95}               % asymmetry amplitude
  \def\profileLambda{50}        % localization strength
  \def\zstar{2.451}             % minimum-radius location for displayed defaults
  \def\smax{0.55}               % max narrowing fraction
  \def\stenosisKernel#1{exp(-\profileLambda*((#1)-\zs+\kapA*exp(-0.5*(((#1)-\za)/\sigmaA)^2))^4)}
  % UPPER WALL
  \draw[wallgray, fill=wallgraylight, line width=0.9pt]
    plot[domain=0:\L, samples=140]
       (\x, {\R*(1 - \smax*\stenosisKernel{\x}) + 0.18})
    -- (\L, \R+0.18) -- (\L,\R+0.36) -- (0,\R+0.36) -- cycle;
  % LOWER WALL
  \draw[wallgray, fill=wallgraylight, line width=0.9pt]
    plot[domain=0:\L, samples=140]
       (\x, {-\R*(1 - \smax*\stenosisKernel{\x}) - 0.18})
    -- (\L, -\R-0.18) -- (\L,-\R-0.36) -- (0,-\R-0.36) -- cycle;
  % LUMEN
  \fill[bloodredlight]
    plot[domain=0:\L, samples=140]
       (\x, {\R*(1 - \smax*\stenosisKernel{\x})})
    -- plot[domain=\L:0, samples=140]
       (\x, {-\R*(1 - \smax*\stenosisKernel{\x})})
    -- cycle;
  \draw[bloodred!70!black, line width=0.5pt]
    plot[domain=0:\L, samples=140]
       (\x, {\R*(1 - \smax*\stenosisKernel{\x})});
  \draw[bloodred!70!black, line width=0.5pt]
    plot[domain=0:\L, samples=140]
       (\x, {-\R*(1 - \smax*\stenosisKernel{\x})});
  % centerline
  \draw[centerline] (-0.2,0) -- (\L+0.2,0);
  % R_base measure
  \draw[<->, mathblue, line width=0.6pt] (0.7,0) -- (0.7,\R)
    node[midway, right=1pt, geomLabel, text=mathblue] {$R_{\mathrm{base}}$};
  % R_min measure (at z_star)
  \pgfmathsetmacro{\Rmin}{\R*(1-\smax*\stenosisKernel{\zstar})}
  \draw[<->, stenosisOrange, line width=0.7pt] (\zstar,0) -- (\zstar,\Rmin)
    node[midway, right=1pt, geomLabel, text=stenosisOrange]
      {$R_{\min}$};
  \node[geomLabel, text=stenosisOrange, anchor=north west]
    at (\zstar+0.18,-\Rmin-0.12)
    {$R_0(z_\star)=R_{\min}=R_{\mathrm{base}}(1-s_{\max})$};
  \draw[stenosisOrange!60, dotted, line width=0.4pt] (\zstar,\R+0.4) -- (\zstar,-\R-0.55);
  \node[font=\scriptsize, stenosisOrange!85!black, fill=white,
        fill opacity=0.88, text opacity=1, inner sep=1pt]
    at (\zstar,-\R-0.68) {$z_\star$};
  % Keep the upper annotation band reserved for pressure taps and pressure drop.
  % proximal / distal pressure taps
  \pgfmathsetmacro{\zp}{\zstar-0.85}
  \pgfmathsetmacro{\zd}{\zstar+1.25}
  \draw[mathblue, line width=0.6pt] (\zp,\R+0.36) -- (\zp,\R+1.08);
  \node[tapLabel, anchor=south east]
    at (\zp-0.06,\R+1.08) {$p_{\mathrm{g},\mathrm{prox}}(t)$};
  \fill[mathblue] (\zp,\R+0.36) circle (1.2pt);
  \draw[mathblue, line width=0.6pt] (\zd,\R+0.36) -- (\zd,\R+1.08);
  \node[tapLabel, anchor=south west]
    at (\zd+0.06,\R+1.08) {$p_{\mathrm{g},\mathrm{dist}}(t)$};
  \fill[mathblue] (\zd,\R+0.36) circle (1.2pt);
  % Delta p arrow
  \draw[<->, mathblue, line width=0.7pt]
    (\zp,\R+1.62) -- (\zd,\R+1.62)
    node[midway, above=1pt, tapLabel]
      {$\Delta p(t)$};
  % flow Q(t) inlet
  \draw[flowArrow, line width=1.4pt] (-1.1,0) -- (-0.1,0)
    node[midway, above, labelB] {$Q(t)$};
  % outlet flow
  \draw[flowArrow, line width=1.0pt] (\L+0.1,0) -- (\L+1.1,0);
  % signed 1D shear proxy callout, kept below the pressure-tap labels
  \draw[->, stenosisOrange, line width=0.55pt]
    (\zstar+1.72,0.55) -- (\zstar+0.18,\Rmin+0.05);
  \node[shearLabel, anchor=west]
    at (\zstar+1.75,0.55) {signed $\tau_{w,z}^{1D}$\\proxy};
  % axes
  \draw[axisArrow] (-1.2,-\R-1.05) -- (\L+1.2,-\R-1.05)
    node[right, labelM] {$z$};
  % tick at z_star (no textual label here; the proximal/distal anchor labels
  % below give axial context, and the lumen geometry already marks z_star)
  \draw[mathblue, line width=0.4pt] (\zstar,-\R-1.05) -- (\zstar,-\R-0.92);
  % proximal / distal labels at base
  \node[font=\scriptsize, mathblue!80!black] at (\zp,-\R-0.85)
    {$z_{\mathrm{prox}}$};
  \node[font=\scriptsize, mathblue!80!black] at (\zd,-\R-0.85)
    {$z_{\mathrm{dist}}$};
\end{tikzpicture}
