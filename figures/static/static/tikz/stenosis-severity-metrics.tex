% ---------------------------------------------------------------
% figures/static/static/tikz/stenosis-severity-metrics.tex
% Compact pgfplots panel of severity-relevant signals:
% Delta p(t), mean Delta p, instantaneous pressure ratio, tau_{w,z}^{1D}, Q(t).
% ---------------------------------------------------------------
\begin{tikzpicture}[fig]
  \pgfplotsset{
    compat=1.18,
    sevpanel/.style={width=5.75cm, height=3.35cm,
      tick label style={font=\scriptsize},
      label style={font=\scriptsize},
      xlabel style={font=\scriptsize},
      ylabel style={font=\scriptsize},
      title style={font=\footnotesize\bfseries, yshift=-1pt},
      xlabel={$t/T_h$}, xmin=0, xmax=1,
      xtick={0,0.5,1},
      major tick length=2pt,
      grid=both, grid style={dashed, gray!25},
      every axis plot/.append style={line width=0.9pt}}
  }

  % top-left: Q(t)
  \begin{axis}[name=ax1, sevpanel, ylabel={$Q$ [mL/s]},
      title={inflow $Q_{\mathrm{in}}(t)$}, ymin=0, ymax=3.2]
    \addplot[mathblue!80!black, smooth, samples=200, domain=0:1]
      {1.0+1.7*max(0, sin(360*x)) + 0.3*max(0,sin(360*x))^3};
  \end{axis}

  % top-right: Delta p(t)
  \begin{axis}[name=ax2, sevpanel,
      at={($(ax1.east)+(1.95cm,0)$)}, anchor=west,
      ylabel={$\Delta p$ [mmHg]},
      title={pressure drop $\Delta p(t)$}, ymin=-2, ymax=42]
    \addplot[bloodred!75!black, smooth, samples=200, domain=0:1]
      {2 + 30*max(0, sin(360*x))^2 + 5*max(0,sin(360*x))};
    \addplot[bloodred!45!black, dashed, samples=2, domain=0:1]
      {12};
    \node[font=\footnotesize, bloodred!70!black, anchor=west,
          fill=white, fill opacity=0.9, text opacity=1, inner sep=1pt]
      at (axis cs:0.66,15.5) {$\overline{\Delta p}_{\mathcal{T}_{\mathrm{HF}}}$};
  \end{axis}

  % bottom-left: instantaneous reference-consistent pressure ratio.
  % No clinical threshold line is drawn; thresholds belong to external
  % measurement/comparison studies, not to the reduced ratio definition.
  \begin{axis}[name=ax3, sevpanel,
      at={($(ax1.south)+(0,-1.75cm)$)}, anchor=north,
      ylabel={$P_{\mathrm{dist}}^{\mathrm{ratio}}(t)/
      P_{\mathrm{prox}}^{\mathrm{ratio}}(t)$},
      title={pressure ratio $\mathrm{PR}_{1D}(t)$},
      ymin=0.5, ymax=1.05, ytick={0.6,0.8,1.0}]
    \addplot[bloodred!80!black, smooth, samples=200, domain=0:1]
      {0.95 - 0.18*max(0, sin(360*x))^2};
  \end{axis}

  % bottom-right: signed 1D axial shear proxy at minimum-area station
  \begin{axis}[name=ax4, sevpanel,
      at={($(ax3.east)+(1.95cm,0)$)}, anchor=west,
      ylabel={$\tau_{w,z}^{1D}$ [Pa]},
      title={signed shear proxy $\tau_{w,z}^{1D}(t)$},
      ymin=-1, ymax=22]
    \addplot[accentTeal!70!black, smooth, samples=200, domain=0:1]
      {1 + 18*max(0, sin(360*x))^2};
  \end{axis}
\end{tikzpicture}
