% ---------------------------------------------------------------
% figures/static/static/tikz/stenosis-parameterization.tex
% Parametric profile R(z;smax)=R_base(1 - smax psi(z)).
% Uses the C^\infty asymmetric kernel in Definition~\ref{def:1d-stenosis-reference-geometry}.
% ---------------------------------------------------------------
\begin{tikzpicture}[fig]
  \begin{axis}[name=ax,
      width=11.2cm, height=6.0cm,
      xlabel={axial coordinate $z$},
      ylabel={reference radius $R_0(z;s_{\max})$},
      xmin=0, xmax=6, ymin=0, ymax=1.25,
      ytick={0,0.5,1.0},
      yticklabels={$0$,$R_{\mathrm{base}}/2$,$R_{\mathrm{base}}$},
      xtick={2.5,3.4},
      xticklabels={$z_a$,$z_s$},
      tick label style={font=\scriptsize},
      xlabel style={font=\footnotesize},
      ylabel style={font=\footnotesize},
      legend style={font=\footnotesize, at={(0.98,0.04)},
                    anchor=south east, draw=mathblue!40, fill=white,
                    fill opacity=0.92, text opacity=1, inner sep=3pt},
      grid=both, grid style={dashed, gray!22}]

    \def\za{2.5}
    \def\zs{3.4}
    \def\sigmaA{1.0}
    \def\kapA{0.95}
    \def\profileLambda{50}
    \def\zstar{2.451}
    \def\stenosisKernel#1{exp(-\profileLambda*((#1)-\zs+\kapA*exp(-0.5*(((#1)-\za)/\sigmaA)^2))^4)}

    % smax = 0.2
    \addplot[mathblue!80!black, line width=1pt, samples=200,
             domain=0:6] {1 - 0.2 * \stenosisKernel{x}};
    \addlegendentry{$s_{\max}=0.2$}

    % smax = 0.35
    \addplot[stenosisOrange, line width=1pt, samples=200,
             domain=0:6] {1 - 0.35 * \stenosisKernel{x}};
    \addlegendentry{$s_{\max}=0.35$}

    % smax = 0.55 (upper end of first solver-screening range)
    \addplot[bloodred!80!black, line width=1.1pt, samples=200,
             domain=0:6] {1 - 0.55 * \stenosisKernel{x}};
    \addlegendentry{$s_{\max}=0.55$}

    % R_min indicator for smax=0.55
    \draw[<->, bloodred, line width=0.5pt] (axis cs:\zstar,0) -- (axis cs:\zstar,0.45)
      node[midway, right=1pt, font=\scriptsize, bloodred,
           fill=white, fill opacity=0.88, text opacity=1, inner sep=1pt]
        {$R_{\min}$};
    % z_star marks the minimum-radius location for the displayed defaults.
    \draw[bloodred!60, dashed, line width=0.5pt] (axis cs:\zstar,0) -- (axis cs:\zstar,1.03)
      node[pos=0.93, left=1pt, font=\scriptsize, bloodred!70!black,
           fill=white, fill opacity=0.9, text opacity=1, inner sep=1pt]
        {$z_\star$};
    \draw[mathblue!55, dotted, line width=0.5pt] (axis cs:\za,0) -- (axis cs:\za,1.03);
    \draw[mathblue!55, dotted, line width=0.5pt] (axis cs:\zs,0) -- (axis cs:\zs,1.03);
  \end{axis}
\end{tikzpicture}
