% Single-vessel notation map. Visual companion to the

% rho, eta, eta_eff, Gamma_w(t), and the outward unit normal n-hat.

\begin{tikzpicture}[fig, scale=1.0]
  % cylinder side view
  \def\L{6.4}
  \def\R{1.0}
  \path[vesselWall] (0,\R+0.25) rectangle (\L,\R);
  \path[vesselWall] (0,-\R-0.25) rectangle (\L,-\R);
  \path[lumen] (0,\R) rectangle (\L,-\R);
  \draw[bloodred!70!black, line width=0.5pt] (0,\R) -- (\L,\R);
  \draw[bloodred!70!black, line width=0.5pt] (0,-\R) -- (\L,-\R);
  \draw[mathblue, line width=0.9pt] (0,\R+0.25) -- (0,-\R-0.25);
  \draw[mathblue, line width=0.9pt] (\L,\R+0.25) -- (\L,-\R-0.25);
  \draw[centerline] (-0.2,0) -- (\L+0.2,0);
  \node[labelM, font=\scriptsize, anchor=north] at (0,-\R-0.28)
    {$S_{\mathrm{in}}(t)$};
  \node[labelM, font=\scriptsize, anchor=north] at (\L,-\R-0.28)
    {$S_{\mathrm{out}}(t)$};

  % R, d_v
  \draw[<->, mathblue, line width=0.6pt] (1.2,0) -- (1.2,\R)
    node[midway, right, labelM] {$R$};
  \draw[<->, mathblue, line width=0.6pt] (4.6,-\R) -- (4.6,\R)
    node[midway, right, labelM] {$d_v=2R$};

  % A: cross section disk inset
  \begin{scope}[shift={(2.0,2.5)}]
    \path[lumen] (0,0) circle (0.6);
    \draw[mathblue, line width=0.7pt] (0,0) circle (0.6);
    \draw[<->, mathblue, line width=0.5pt] (0,0) -- (0.6,0)
      node[midway, above, labelM, font=\scriptsize] {$R$};
    \node[labelM, font=\scriptsize, below=1pt] at (0,-0.6)
      {$A=\pi R^2$};
  \end{scope}
  \draw[->, dashed, mathblue!70!black] (2.0,1.85) -- (1.5,\R+0.05);

  % Q label
  \draw[flowArrow, line width=1.2pt] (-0.9,0) -- (-0.1,0);
  \node[labelB, above=2pt] at (-0.5,0)
    {$Q=\!\!\int_{S} u_z\,dS$};

  % u(t,x) sample
  \coordinate (P) at (3.2,0.3);
  \fill[bloodred] (P) circle (1.1pt);
  \draw[flowArrow] (P) -- ++(0.7,0.05) node[above, labelB] {$\vu(t,\vx)$};

  % p(t,x)
  \node[labelM] at (3.2,-0.7) {$p(t,\vx)$};

  % wall normal
  \coordinate (Pw) at (5.6,\R);
  \fill[mathblue] (Pw) circle (1pt);
  \draw[axisArrow] (Pw) -- ++(0,0.55) node[above, labelM] {$\nhat$};
  \node[labelW, font=\scriptsize, above right=2pt and 4pt] at (Pw) {$\Gamma_w(t)$};

  % material params (compact boxed inset)
  \node[draw=mathblue!55, fill=mathblueLite!50, rounded corners=2pt,
        inner sep=3.5pt, font=\footnotesize, align=left, anchor=north west]
    at (\L-0.65,2.45) {%
      \textbf{material}\\
      $\rho$\;density\\
      $\eta$\;dyn.\ visc.\\
      $\eta_{\mathrm{eff}}$\;eff.\ visc.\\
      $\nu=\eta/\rho$};
  % flow indicator
  \draw[flowArrow, line width=1.1pt] (\L+0.1,0) -- (\L+0.78,0);
  \node[labelB, above=2pt] at (\L+0.44,0) {flow};

  % axes
  \draw[axisArrow] (-1.12,-\R-0.72) -- (-1.12,\R+0.6) node[above, labelM] {$r$};
  \draw[axisArrow] (-1.12,-\R-0.72) -- (\L+0.2,-\R-0.72) node[right, labelM] {$z$};
