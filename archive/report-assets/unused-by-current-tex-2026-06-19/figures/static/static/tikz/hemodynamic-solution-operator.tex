% ---------------------------------------------------------------
% Forward-map schematic. Three cards (input
% descriptor -> map -> outputs), uniform inner padding, a
% single schemEdge weight, no in-figure cautionary note (the
% caption already states it is a target specification, not
% accuracy evidence).
% ---------------------------------------------------------------
\begin{tikzpicture}[fig, scale=1.06, transform shape,
    node distance=12mm,
    card/.style={draw=mathblue!65!black, rounded corners=3pt, line width=0.8pt,
                 fill=mathblueLite!32, text width=36mm, minimum height=30mm,
                 align=left, inner sep=5pt, font=\footnotesize},
    opcard/.style={draw=accentTeal!75!black, rounded corners=4pt, line width=1pt,
                   fill=accentTealLite!60, text width=34mm, minimum height=30mm,
                   align=center, inner sep=5pt, font=\footnotesize},
    outcard/.style={draw=bloodred!70!black, rounded corners=3pt, line width=0.8pt,
                    fill=bloodredlight!50, text width=38mm, minimum height=30mm,
                    align=left, inner sep=5pt, font=\footnotesize},
    link/.style={schemEdge, line width=1.0pt}]

  \node[card] (x)
    {\textbf{descriptor}\\[2pt]
     $a_{1D}\in\mathcal X_{1D}$:\\
     $R_{\mathrm{base}},s,\beta,\rho_0,\mathcal R$\\
     $Q_{\mathrm{in}},p_{\mathrm{out},g}$\\
     \textbf{derived geometry}\\
     $R_0=R_{\mathrm{base}}(1-s)$\\
     $A_0=\pi R_0^2$\\[1pt]
     \textbf{output specs}\\
     $\pi_{\Delta p}$, $\pi_{\mathrm{PR}}$};

  \node[opcard, right=of x] (g)
    {{\normalsize$\mathcal G_{1D}^{\Delta p}$}\\[-1pt]
     {\scriptsize ratio extension $\mathcal G_{1D}^{\mathrm{PR}}$}\\[1pt]
     \rule{18mm}{0.35pt}\\[1pt]
     forward-map\\
     formulation};

  \node[outcard, right=of g] (y)
    {\textbf{outputs $\mathcal Y$}\\[2pt]
     returned fields: $A,Q,p_{\mathrm{g}}$\\
     reduced shear: $\tau_{w,z}^{1D}$\\
     pressure drop: $\Delta p(t)$\\
     pressure ratio: $\mathrm{PR}_{1D}$ when admissible};

  \draw[link] (x.east) -- (g.west);
  \draw[link, bloodred!70!black] (g.east) -- (y.west);
\end{tikzpicture}
