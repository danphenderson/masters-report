% report/assets/tikz/lagrangian-eulerian-map.tex
% Lagrangian / Eulerian flow-map schematic.
% Shows L_t : Omega_B(0) -> Omega_B(t), material point xi,
% current point x(t; xi), trajectory T_xi, and Eulerian
% velocity u(t,x) at the current configuration.
% ---------------------------------------------------------------
\begin{tikzpicture}[fig, scale=1.06, transform shape]
  % LEFT: reference configuration Omega_B(0)
  \begin{scope}[local bounding box=L]
    \path[draw=mathblue, line width=0.9pt, fill=mathblueLite!50, rounded corners=8pt]
      (-1.8,-1.1) .. controls (-1.6,1.1) and (1.5,1.5) .. (1.9,0.4)
      .. controls (2.2,-0.7) and (1.0,-1.5) .. (-0.4,-1.4)
      .. controls (-1.4,-1.35) and (-1.9,-1.25) .. (-1.8,-1.1) -- cycle;
    \node[panelTitle] at (0,2.05) {(a) material label};
    \node[tagLabelM] at (0,1.58) {$\OmgB(0)$};
    \coordinate (xi) at (-0.6,0.1);
    \fill[mathblue] (xi) circle (1.5pt);
    \node[tagLabelM, below=2pt] at (xi) {$\vxi$};
  \end{scope}

  % RIGHT: current configuration Omega_B(t)
  \begin{scope}[shift={(7,-0.2)}, local bounding box=R]
    \path[draw=bloodred!70!black, line width=0.9pt,
          fill=bloodredlight!70, rounded corners=10pt]
      (-2.0,-0.9) .. controls (-1.5,1.4) and (1.5,1.6) .. (2.2,0.3)
      .. controls (2.6,-1.0) and (0.8,-1.5) .. (-0.6,-1.6)
      .. controls (-1.6,-1.55) and (-2.1,-1.1) .. (-2.0,-0.9) -- cycle;
    \node[panelTitle, bloodred!70!black] at (0,2.05)
      {(b) Eulerian position at time $t$};
    \node[tagLabelB] at (0,1.58) {$\OmgBt$};
    \coordinate (x) at (0.7,0.4);
    \fill[bloodred!70!black] (x) circle (1.5pt);
    \node[tagLabelB, below=2pt] at (x) {$\vx(t;\vxi)$};
    % Eulerian velocity
    \draw[flowArrow] (x) -- ++(0.95,0.25)
      node[above right, tagLabelB, xshift=-3pt] {$\vu(t,\vx)$};
  \end{scope}

  % L_t arrow between the two domains
  \draw[->, mathblue!80!black, line width=1.1pt]
    (L.east) to[bend left=14]
      node[above, tagLabelM] {$\La_t$}
      node[below, tagLabelM, font=\scriptsize] {$\vxi\mapsto\vx(t;\vxi)$}
    (R.west);

  % Trajectory in space-time below the two domains
  \begin{scope}[shift={(0,-3.7)}]
    \node[panelTitle] at (3.0,1.95) {(c) particle trajectory};
    \draw[axisArrow] (-2.2,0) -- (8.2,0) node[right, labelM] {$s$};
    \draw[axisArrow] (-2.2,0) -- (-2.2,1.55) node[above, labelM] {$\vx$};
    % trajectory curve
    \draw[bloodred!70!black, line width=1pt, smooth]
      plot[domain=-2.0:7.5, samples=60]
        (\x, {0.3 + 0.45*sin(40*(\x+2)) + 0.04*(\x+2)*(\x+2)/8});
    % start marker (xi)
    \fill[mathblue] (-2.0,0.3) circle (1.4pt);
    \node[tagLabelM, below=2pt] at (-2.0,0.3) {$\vxi$};
    \node[tagLabelM, font=\scriptsize, above=2pt] at (-1.48,0.55)
      {$s=0$};
    % end marker
    \fill[bloodred!70!black] (7.5,1.0) circle (1.4pt);
    \node[tagLabelB, right=2pt] at (7.5,1.0) {$\vx(t;\vxi)$};
    \node[tagLabelB, font=\scriptsize, below=2pt] at (7.5,0.68) {$s=t$};
    % trajectory label
    \node[tagLabelB, font=\scriptsize\itshape] at (3.0,1.43)
      {$T_{\vxi}$};
  \end{scope}
\end{tikzpicture}
